function openProxy() {
   export https_proxy=http://127.0.0.1:7890
   export http_proxy=http://127.0.0.1:7890
   export all_proxy=socks5://127.0.0.1:7890
   echo -e "free on termial"
}

function closeProxy() {
   unset https_proxy
   unset http_proxy
   unset all_proxy
   echo -e "proxy closed"
}
